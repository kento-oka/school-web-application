CREATE TABLE login_user(
    id INTEGER AUTO_INCREMENT,
    signin_id VARCHAR(255) NOT NULL UNIQUE,
    signin_pw VARCHAR(255) NOT NULL,
    name TEXT NOT NULL,
    type TEXT NOT NULL,
    created_at DATETIME NOT NULL,
    deleted_at DATETIME DEFAULT NULL,
    PRIMARY KEY(id)
);
